Class Hierarchy


Button



						class Button extends UI5Element {

							static get metadata() {} // properties, events and slots
							static get styles() {} // styles
							static get render() {} // rendering engine
							static get template() {} // template for rendering

							onBeforeRendering() {}
							onAfterRendering() {}
							onEnterDOM() {}
							onExitDOM() {}
						}
